% This file contains the scalar version of the equations:
%   C = m1·w^(1) + m2·w^(2) + ... + m48·w^(48)

%   t00_i   = Im(m1)

%   t01_i   = Im(m2)
%   t22_r   = Re(m23)
%   t22_i   = Im(m23)
%   t47_r   = Re(m48)
%   t47_i   = Im(m48)
%	 
% The output is the 4x4 complex valued C matrix.
